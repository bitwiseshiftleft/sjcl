  8 /** @namespace Random number generator
  9  *
 10  * @description
 11  * <p>
 12  * This random number generator is a derivative of Ferguson and Schneier's
 13  * generator Fortuna.  It collects entropy from various events into several
 14  * pools, implemented by streaming SHA-256 instances.  It differs from
 15  * ordinary Fortuna in a few ways, though.
 16  * </p>
 17  *
 18  * <p>
 19  * Most importantly, it has an entropy estimator.  This is present because
 20  * there is a strong conflict here between making the generator available
 21  * as soon as possible, and making sure that it doesn't "run on empty".
 22  * In Fortuna, there is a saved state file, and the system is likely to have
 23  * time to warm up.
 24  * </p>
 25  *
 26  * <p>
 27  * Second, because users are unlikely to stay on the page for very long,
 28  * and to speed startup time, the number of pools increases logarithmically:
 29  * a new pool is created when the previous one is actually used for a reseed.
 30  * This gives the same asymptotic guarantees as Fortuna, but gives more
 31  * entropy to early reseeds.
 32  * </p>
 33  *
 34  * <p>
 35  * The entire mechanism here feels pretty klunky.  Furthermore, there are
 36  * several improvements that should be made, including support for
 37  * dedicated cryptographic functions that may be present in some browsers;
 38  * state files in local storage; cookies containing randomness; etc.  So
 39  * look for improvements in future versions.
 40  * </p>
 41  */
 42 sjcl.random = {
 43   /** Generate several random words, and return them in an array
 44    * @param {Number} nwords The number of words to generate.
 45    */
 46   randomWords: function (nwords, paranoia) {
 47     var out = [], i, readiness = this.isReady(paranoia), g;
 48   
 49     if (readiness === this._NOT_READY) {
 50       throw new sjcl.exception.notready("generator isn't seeded");
 51     } else if (readiness & this._REQUIRES_RESEED) {
 52       this._reseedFromPools(!(readiness & this._READY));
 53     }
 54   
 55     for (i=0; i<nwords; i+= 4) {
 56       if ((i+1) % this._MAX_WORDS_PER_BURST === 0) {
 57         this._gate();
 58       }
 59    
 60       g = this._gen4words();
 61       out.push(g[0],g[1],g[2],g[3]);
 62     }
 63     this._gate();
 64   
 65     return out.slice(0,nwords);
 66   },
 67   
 68   setDefaultParanoia: function (paranoia) {
 69     this._defaultParanoia = paranoia;
 70   },
 71   
 72   /**
 73    * Add entropy to the pools.
 74    * @param data The entropic value.  Should be a 32-bit integer, array of 32-bit integers, or string
 75    * @param {Number} estimatedEntropy The estimated entropy of data, in bits
 76    * @param {String} source The source of the entropy, eg "mouse"
 77    */
 78   addEntropy: function (data, estimatedEntropy, source) {
 79     source = source || "user";
 80   
 81     var id,
 82       i, ty = 0, tmp,
 83       t = (new Date()).valueOf(),
 84       robin = this._robins[source],
 85       oldReady = this.isReady();
 86       
 87     id = this._collectorIds[source];
 88     if (id === undefined) { id = this._collectorIds[source] = this._collectorIdNext ++; }
 89       
 90     if (robin === undefined) { robin = this._robins[source] = 0; }
 91     this._robins[source] = ( this._robins[source] + 1 ) % this._pools.length;
 92   
 93     switch(typeof(data)) {
 94       
 95     case "number":
 96       data=[data];
 97       ty=1;
 98       break;
 99       
100     case "object":
101       if (estimatedEntropy === undefined) {
102        /* horrible entropy estimator */
103        estimatedEntropy = 0;
104        for (i=0; i<data.length; i++) {
105           tmp= data[i];
106           while (tmp>0) {
107            estimatedEntropy++;
108            tmp = tmp >>> 1;
109           }
110         }
111       }
112       this._pools[robin].update([id,this._eventId++,ty||2,estimatedEntropy,t,data.length].concat(data));
113       break;
114       
115     case "string":
116       if (estimatedEntropy === undefined) {
117        /* English text has just over 1 bit per character of entropy.
118         * But this might be HTML or something, and have far less
119         * entropy than English...  Oh well, let's just say one bit.
120         */
121        estimatedEntropy = data.length;
122       }
123       this._pools[robin].update([id,this._eventId++,3,estimatedEntropy,t,data.length]);
124       this._pools[robin].update(data);
125       break;
126       
127     default:
128       
129       throw new sjcl.exception.bug("random: addEntropy only supports number, array or string");
130     }
131   
132     /* record the new strength */
133     this._poolEntropy[robin] += estimatedEntropy;
134     this._poolStrength += estimatedEntropy;
135   
136     /* fire off events */
137     if (oldReady === this._NOT_READY) {
138       if (this.isReady() !== this._NOT_READY) {
139         this._fireEvent("seeded", Math.max(this._strength, this._poolStrength));
140       }
141       this._fireEvent("progress", this.getProgress());
142     }
143   },
144   
145   /** Is the generator ready? */
146   isReady: function (paranoia) {
147     var entropyRequired = this._PARANOIA_LEVELS[ (paranoia !== undefined) ? paranoia : this._defaultParanoia ];
148   
149     if (this._strength && this._strength >= entropyRequired) {
150       return (this._poolEntropy[0] > this._BITS_PER_RESEED && (new Date()).valueOf() > this._nextReseed) ?
151         this._REQUIRES_RESEED | this._READY :
152         this._READY;
153     } else {
154       return (this._poolStrength >= entropyRequired) ?
155         this._REQUIRES_RESEED | this._NOT_READY :
156         this._NOT_READY;
157     }
158   },
159   
160   /** Get the generator's progress toward readiness, as a fraction */
161   getProgress: function (paranoia) {
162     var entropyRequired = this._PARANOIA_LEVELS[ paranoia ? paranoia : this._defaultParanoia ];
163   
164     if (this._strength >= entropyRequired) {
165       return 1.0;
166     } else {
167       return (this._poolStrength > entropyRequired) ?
168         1.0 :
169         this._poolStrength / entropyRequired;
170     }
171   },
172   
173   /** start the built-in entropy collectors */
174   startCollectors: function () {
175     if (this._collectorsStarted) { return; }
176   
177     if (window.addEventListener) {
178       window.addEventListener("load", this._loadTimeCollector, false);
179       window.addEventListener("mousemove", this._mouseCollector, false);
180     } else if (document.attachEvent) {
181       document.attachEvent("onload", this._loadTimeCollector);
182       document.attachEvent("onmousemove", this._mouseCollector);
183     }
184     else {
185       throw new sjcl.exception.bug("can't attach event");
186     }
187   
188     this._collectorsStarted = true;
189   },
190   
191   /** stop the built-in entropy collectors */
192   stopCollectors: function () {
193     if (!this._collectorsStarted) { return; }
194   
195     if (window.removeEventListener) {
196       window.removeEventListener("load", this._loadTimeCollector);
197       window.removeEventListener("mousemove", this._mouseCollector);
198     } else if (window.detachEvent) {
199       window.detachEvent("onload", this._loadTimeCollector);
200       window.detachEvent("onmousemove", this._mouseCollector);
201     }
202     this._collectorsStarted = false;
203   },
204   
205   /* use a cookie to store entropy.
206   useCookie: function (all_cookies) {
207       throw new sjcl.exception.bug("random: useCookie is unimplemented");
208   },*/
209   
210   /** add an event listener for progress or seeded-ness. */
211   addEventListener: function (name, callback) {
212     this._callbacks[name][this._callbackI++] = callback;
213   },
214   
215   /** remove an event listener for progress or seeded-ness */
216   removeEventListener: function (name, cb) {
217     var i, j, cbs=this._callbacks[name], jsTemp=[];
218   
219     /* I'm not sure if this is necessary; in C++, iterating over a
220      * collection and modifying it at the same time is a no-no.
221      */
222   
223     for (j in cbs) {
224      if (cbs.hasOwnProperty[j] && cbs[j] === cb) {
225         jsTemp.push(j);
226       }
227     }
228   
229     for (i=0; i<jsTemp.length; i++) {
230       j = jsTemp[i];
231       delete cbs[j];
232     }
233   },
234   
235   /* private */
236   _pools                   : [new sjcl.hash.sha256()],
237   _poolEntropy             : [0],
238   _reseedCount             : 0,
239   _robins                  : {},
240   _eventId                 : 0,
241   
242   _collectorIds            : {},
243   _collectorIdNext         : 0,
244   
245   _strength                : 0,
246   _poolStrength            : 0,
247   _nextReseed              : 0,
248   _key                     : [0,0,0,0,0,0,0,0],
249   _counter                 : [0,0,0,0],
250   _cipher                  : undefined,
251   _defaultParanoia         : 6,
252   
253   /* event listener stuff */
254   _collectorsStarted       : false,
255   _callbacks               : {progress: {}, seeded: {}},
256   _callbackI               : 0,
257   
258   /* constants */
259   _NOT_READY               : 0,
260   _READY                   : 1,
261   _REQUIRES_RESEED         : 2,
262 
263   _MAX_WORDS_PER_BURST     : 65536,
264   _PARANOIA_LEVELS         : [0,48,64,96,128,192,256,384,512,768,1024],
265   _MILLISECONDS_PER_RESEED : 30000,
266   _BITS_PER_RESEED         : 80,
267   
268   /** Generate 4 random words, no reseed, no gate.
269    * @private
270    */
271   _gen4words: function () {
272     for (var i=0; i<4; i++) {
273       this._counter[i] = this._counter[i]+1 | 0;
274       if (this._counter[i]) { break; }
275     }
276     return this._cipher.encrypt(this._counter);
277   },
278   
279   /* Rekey the AES instance with itself after a request, or every _MAX_WORDS_PER_BURST words.
280    * @private
281    */
282   _gate: function () {
283     this._key = this._gen4words().concat(this._gen4words());
284     this._cipher = new sjcl.cipher.aes(this._key);
285   },
286   
287   /** Reseed the generator with the given words
288    * @private
289    */
290   _reseed: function (seedWords) {
291     this._key = sjcl.hash.sha256.hash(this._key.concat(seedWords));
292     this._cipher = new sjcl.cipher.aes(this._key);
293     for (var i=0; i<4; i++) {
294       this._counter[i] = this._counter[i]+1 | 0;
295       if (this._counter[i]) { break; }
296     }
297   },
298   
299   /** reseed the data from the entropy pools
300    * @param full If set, use all the entropy pools in the reseed.
301    */
302   _reseedFromPools: function (full) {
303     var reseedData = [], strength = 0, i;
304   
305     this._nextReseed = reseedData[0] =
306       (new Date()).valueOf() + this._MILLISECONDS_PER_RESEED;
307     
308     for (i=0; i<16; i++) {
309       /* On some browsers, this is cryptographically random.  So we might
310        * as well toss it in the pot and stir...
311        */
312       reseedData.push(Math.random()*0x100000000|0);
313     }
314     
315     for (i=0; i<this._pools.length; i++) {
316      reseedData = reseedData.concat(this._pools[i].finalize());
317      strength += this._poolEntropy[i];
318      this._poolEntropy[i] = 0;
319    
320      if (!full && (this._reseedCount & (1<<i))) { break; }
321     }
322   
323     /* if we used the last pool, push a new one onto the stack */
324     if (this._reseedCount >= 1 << this._pools.length) {
325      this._pools.push(new sjcl.hash.sha256());
326      this._poolEntropy.push(0);
327     }
328   
329     /* how strong was this reseed? */
330     this._poolStrength -= strength;
331     if (strength > this._strength) {
332       this._strength = strength;
333     }
334   
335     this._reseedCount ++;
336     this._reseed(reseedData);
337   },
338   
339   _mouseCollector: function (ev) {
340     var x = ev.x || ev.clientX || ev.offsetX, y = ev.y || ev.clientY || ev.offsetY;
341     sjcl.random.addEntropy([x,y], 2, "mouse");
342   },
343   
344   _loadTimeCollector: function (ev) {
345     var d = new Date();
346     sjcl.random.addEntropy(d, 2, "loadtime");
347   },
348   
349   _fireEvent: function (name, arg) {
350     var j, cbs=sjcl.random._callbacks[name], cbsTemp=[];
351     /* TODO: there is a race condition between removing collectors and firing them */ 
352 
353     /* I'm not sure if this is necessary; in C++, iterating over a
354      * collection and modifying it at the same time is a no-no.
355      */
356   
357     for (j in cbs) {
358      if (cbs.hasOwnProperty(j)) {
359         cbsTemp.push(cbs[j]);
360      }
361     }
362   
363     for (j=0; j<cbsTemp.length; j++) {
364      cbsTemp[j](arg);
365     }
366   }
367 };
